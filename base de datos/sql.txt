---- procedimiento almacenado buscar usuariario y contraseña
CREATE OR REPLACE FUNCTION 
buscar_usuario(usu varchar,contra varchar) 
RETURNS  usuarios AS $$
DECLARE
n1 usuarios;
BEGIN
SELECT * into n1 from usuarios where "contraseña" = contra and usu = usuario;
return n1;
end;
$$ LANGUAGE plpgsql;
